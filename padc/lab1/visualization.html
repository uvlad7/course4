require 'matplotlib/iruby'
require 'color-generator'
require 'chroma'

Matplotlib::IRuby.activate
plt = Matplotlib::Pyplot

generator = ColorGenerator.new saturation: 1.0, value: 1.0, seed: 12345

processes_number.each do |prc|
    data.select { |r| r[0] == prc }.group_by { |r| r[1..3] }.each do |k, v_arr|
        plt.figure
        plt.gcf.set_size_inches(12, 12)
        plt.title("Number of processes: #{prc}, sizes: #{k}")
        plt.xlabel("Block size")
        plt.ylabel("Time in sec")
        color = generator.create_hex
        et = v_arr.find { |r| r[4] == -1 }[-1]
        plt.axhline(et, zorder: 1, label: "Etalon", color: "##{color}")
        d = [{:x=>[], :y=>[]}, {:x=>[], :y=>[]}, {:x=>[], :y=>[]}]
        v_arr.each do |r|
            next if r[4] == -1
            d[r[4]][:x].push(r[-2])
            d[r[4]][:y].push(r[-1])
        end
        d.each_with_index do |p, i|
            c = color.paint.spin(90 * (i + 1)).to_hex
            plt.axhline(p[:y][p[:x].index(0)], zorder: 0, color: 'black')
            plt.plot(p[:x], p[:y], zorder: 2 * i + 2, label: "Parallelisation: #{parallelisation[i]}", color: c)
            plt.scatter(p[:x], p[:y], 10, zorder: 2 * i + 3, color: "black")
        end
        plt.legend(loc: "upper center", bbox_to_anchor: [0.5, -0.05], fancybox: true, shadow: true, ncol: 4)
        plt.grid
        plt.show
    end
end

processes_number.each do |prc|
    selected = data.select { |r| r[0] == prc }.group_by { |r| r[4] }
    etalons = selected[-1]
    selected.each do |k, v_arr|
        next if k == -1
        plt.figure
        plt.gcf.set_size_inches(12, 12)
        plt.title("Number of processes: #{prc}, parallelisation: #{parallelisation[k]}")
        plt.xlabel("Block size")
        plt.ylabel("Time in sec")
        color = generator.create_hex
        d = {}
        v_arr.each do |r|
            k = r[1..3].inspect
            next if r[4] == -1
            d[k] ||= {}
            d[k][:x] ||= []
            d[k][:x].push(r[-2])
            d[k][:y] ||= []
            d[k][:y].push(r[-1])
        end
        d.each_with_index do |(k, v), i|
            et = etalons.find { |r| r[1..3].inspect == k }[-1]
            c = color.paint.spin(90 * (i + 1)).to_hex
            plt.axhline(et, zorder: 4 * i + 1, color: c)
            plt.axhline(v[:y][v[:x].index(0)], zorder: 0, color: "black")
            plt.plot(v[:x], v[:y], zorder: 4 * i + 2, label: "Size: #{k}", color: c)
            plt.scatter(v[:x], v[:y], 10, zorder: 4 * i + 3, color: "black")
        end
        plt.legend(loc: "upper center", bbox_to_anchor: [0.5, -0.05], fancybox: true, shadow: true, ncol: 4)
        plt.grid
        plt.show
    end
end
